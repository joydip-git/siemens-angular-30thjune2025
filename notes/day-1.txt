class Logic{
	public bool IsEven(int num) => num %2 == 0;
	public static bool IsOdd(int num) => num %2 != 0;

}

Main(){
	Logic logic = new Logic();
	LogicDel evenDel = new LogicDel(logic.IsEven);

	LogicDel oddDel = Logic.IsOdd;

	evenDel(12);
	oddDel.Invoke(13);
}
